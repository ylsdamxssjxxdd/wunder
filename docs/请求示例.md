# 请求示例

本文提供 /wunder 接口的 Python 请求示例，包含最简调用与完整对话流程。

## 0. 依赖与约定

- Python 依赖：`pip install requests`
- 服务地址示例：`http://127.0.0.1:8000/wunder`（请按实际部署修改）

## 1. 最简 /wunder 请求（非流式）

```python
import requests

# /wunder 入口地址，默认本地开发端口
BASE_URL = "http://127.0.0.1:8000/wunder"

# 构造最简请求体：user_id 与 question 为必填
payload = {
    "user_id": "u001",
    "question": "你好",
    "stream": False,  # 关闭流式，直接返回完整答案
}

# 发送请求并检查 HTTP 状态码
resp = requests.post(BASE_URL, json=payload, timeout=60)
resp.raise_for_status()

# 解析 JSON 响应，包含 session_id 与 answer
data = resp.json()
print(data["session_id"])
print(data["answer"])
```

## 2. 完整智能体对话流程（获取工具 + 流式响应）

```python
import json
import requests

# 统一配置服务地址，避免多处硬编码
BASE_HOST = "http://127.0.0.1:8000"
TOOLS_URL = f"{BASE_HOST}/wunder/tools"
WUNDER_URL = f"{BASE_HOST}/wunder"

# 1) 先获取服务端可用工具清单
tools_resp = requests.get(TOOLS_URL, timeout=30)
tools_resp.raise_for_status()
tools_payload = tools_resp.json()

# 2) 从各类工具清单中抽取工具名称，并进行去重
def _append_tools(target, items):
    for item in items:
        if not isinstance(item, dict):
            continue
        name = str(item.get("name", "")).strip()
        if name:
            target.append(name)

tool_names_raw = []
_append_tools(tool_names_raw, tools_payload.get("builtin_tools", []))
_append_tools(tool_names_raw, tools_payload.get("mcp_tools", []))
_append_tools(tool_names_raw, tools_payload.get("skills", []))
_append_tools(tool_names_raw, tools_payload.get("knowledge_tools", []))

seen = set()
tool_names = []
for name in tool_names_raw:
    if name in seen:
        continue
    seen.add(name)
    tool_names.append(name)

# 3) 组装 /wunder 请求，并附带全部工具名称
payload = {
    "user_id": "u001",
    "question": "请基于可用工具完成任务，并说明过程。",
    "tool_names": tool_names,
    "stream": True,
}

# 4) 以 SSE 流式接收响应，逐条解析事件
with requests.post(WUNDER_URL, json=payload, stream=True, timeout=120) as resp:
    resp.raise_for_status()
    current_event = None
    for line in resp.iter_lines(decode_unicode=True):
        if not line:
            continue
        if line.startswith("event:"):
            current_event = line.split(":", 1)[1].strip()
            continue
        if not line.startswith("data:"):
            continue
        # data 行是 JSON 字符串，包含 type/session_id/data/timestamp
        event = json.loads(line.split(":", 1)[1].strip())
        event_type = current_event or event.get("type")
        event_data = event.get("data", {})
        print(event_type, event_data)
        if event_type == "final":
            break
```
